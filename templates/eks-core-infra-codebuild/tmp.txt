



export PROTON_ENV=$(cat proton-inputs-example.json | jq '.environment.name' -r)
export AWS_REGION=$(cat proton-inputs-example.json | jq '.environment.inputs.aws_region' -r)
export TF_STATE_BUCKET=$(cat proton-inputs-example.json | jq '.environment.inputs.tf_state_bucket' -r)
export TF_STATE_BUCKET_REGION=$(cat proton-inputs-example.json | jq '.environment.inputs.tf_state_bucket_region' -r)


terraform init -backend-config="bucket=${TF_STATE_BUCKET}" -backend-config="key=${PROTON_ENV}.tfstate" -backend-config="region=${TF_STATE_BUCKET_REGION}" -upgrade -reconfigure
terraform destroy -var-file=proton-inputs-example.json -var="aws_region=${AWS_REGION}" -auto-approve
